⚙️ Connect to Google Drive

One-time setup (~3 minutes). Your recipes are saved as a single JSON file in your own Google Drive — no cell limits, atomic saves, always reliable.

1
Go to script.google.com → click New project → name it RecipeNest.
2
Delete any code in the editor and paste this: 
// ── SET YOUR OWN SECRET TOKEN ────────────────────────────────
const SECRET = '__YOUR_TOKEN__';

const FILENAME = 'recipenest-data.json';
const COST_FILENAME = 'recipenest-costing.json';

function auth(e) {
  return (e.parameter.token || '') === SECRET;
}

function getOrCreateFile(filename) {
  const files = DriveApp.getFilesByName(filename);
  if (files.hasNext()) return files.next();
  return DriveApp.createFile(filename, '{}', MimeType.PLAIN_TEXT);
}

function readJson(filename) {
  try {
    const file = getOrCreateFile(filename);
    const text = file.getBlob().getDataAsString();
    return text ? JSON.parse(text) : {};
  } catch(e) { return {}; }
}

function writeJson(filename, data) {
  const files = DriveApp.getFilesByName(filename);
  const text  = JSON.stringify(data);
  if (files.hasNext()) {
    files.next().setContent(text);
  } else {
    DriveApp.createFile(filename, text, MimeType.PLAIN_TEXT);
  }
}

function doGet(e) {
  if (!auth(e)) return err('Unauthorized');
  try {
    // Handle POST-via-GET fallback (browser redirect strips POST body)
    if (e.parameter.method === 'POST' && e.parameter.data) {
      const fakeE = { postData: { contents: decodeURIComponent(e.parameter.data) }, parameter: e.parameter };
      return doPost(fakeE);
    }
    if (e.parameter.action === 'loadCost') {
      const data = readJson(COST_FILENAME);
      return ok({ records: data.records || [] });
    }
    const data = readJson(FILENAME);
    return ok({
      recipes:     data.recipes     || [],
      groceryList: data.groceryList || [],
      planner:     data.planner     || {},
      categories:  data.categories  || [],
      savedAt:     data.savedAt     || null
    });
  } catch(e) { return err(e.message); }
}

function doPost(e) {
  if (!auth(e)) return err('Unauthorized');
  try {
    const d = JSON.parse(e.postData.contents);
    if (e.parameter.action === 'saveCost') {
      writeJson(COST_FILENAME, { records: d.records || [] });
      return ok({ saved: true });
    }
    if (e.parameter.action === 'savePhoto') {
      return savePhoto(e, d);
    }
    // Save entire collection atomically as one JSON file
    writeJson(FILENAME, {
      recipes:     d.recipes     || [],
      groceryList: d.groceryList || [],
      planner:     d.planner     || {},
      categories:  d.categories  || [],
      savedAt:     new Date().toISOString()
    });
    return ok({ saved: true, count: (d.recipes || []).length });
  } catch(e) { return err(e.message); }
}

function savePhoto(e, d) {
  const recipeId = d.recipeId || 'photo';
  const mimeType = d.mimeType || 'image/jpeg';
  const ext      = mimeType.split('/')[1] || 'jpg';
  const filename = 'recipenest-photo-' + recipeId + '.' + ext;
  const blob     = Utilities.newBlob(Utilities.base64Decode(d.data), mimeType, filename);
  const existing = DriveApp.getFilesByName(filename);
  while (existing.hasNext()) existing.next().setTrashed(true);
  const file = DriveApp.createFile(blob);
  file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
  const url = 'https://drive.google.com/uc?export=view&id=' + file.getId();
  return ok({ url: url });
}
function ok(data)  { return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); }
function err(msg)  { return ContentService.createTextOutput(JSON.stringify({error: msg})).setMimeType(ContentService.MimeType.JSON); }
3
Enter your secret token in the field below, then click copy above — your token is embedded automatically.
4
Click Deploy → New deployment → gear icon → Web app.
Set Execute as: Me · Who has access: Anyone → Deploy.
Your token keeps data private — only someone with the token can read or write.
